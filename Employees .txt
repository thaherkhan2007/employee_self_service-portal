    async function submitRequest(e, type) {
      e.preventDefault();
      
      if (!currentUser || !window.dataSdk) return;
      
      if (allRequests.length >= 999) {
        showToast('Maximum request limit reached. Please contact support.');
        return;
      }


      let details = '';
      if (type === 'Leave Request') {
        details = `${document.getElementById('leave-type').value}: ${document.getElementById('leave-from').value} to ${document.getElementById('leave-to').value} - ${document.getElementById('leave-reason').value}`;
      } else if (type === 'IT Ticket') {
        details = `${document.getElementById('it-category').value} (${document.getElementById('it-priority').value}): ${document.getElementById('it-description').value}`;
      } else if (type === 'Travel Request') {
        details = `${document.getElementById('travel-dest').value}: ${document.getElementById('travel-from').value} to ${document.getElementById('travel-to').value} - ${document.getElementById('travel-purpose').value}`;
      } else if (type === 'Asset Request') {
        details = `${document.getElementById('asset-type').value}: ${document.getElementById('asset-reason').value}`;
      } else if (type === 'Facility Request') {
        details = `${document.getElementById('facility-type').value}: ${document.getElementById('facility-desc').value}`;
      } else if (type === 'Password Reset') {
        details = `${document.getElementById('reset-system').value}: ${document.getElementById('reset-reason').value}`;
      } else if (type === 'Access Request') {
        details = `${document.getElementById('access-system').value} (${document.getElementById('access-type').value}): ${document.getElementById('access-reason').value}`;
      }


      const requestData = {
        id: 'REQ-' + Date.now().toString(36).toUpperCase(),
        employee_id: currentUser.id,
        employee_name: currentUser.name,
        request_type: type,
        status: 'Pending',
        details: details,
        created_at: new Date().toISOString(),
        updated_at: new Date().toISOString()
      };


      const result = await window.dataSdk.create(requestData);
      if (result.isOk) {
        showToast('Request submitted successfully!');
        closeModal();
      } else {
        showToast('Failed to submit request. Please try again.');
      }
    }

    async function updateRequestStatus(backendId, newStatus) {
      if (!window.dataSdk) return;
      
      const request = allRequests.find(r => r.__backendId === backendId);
      if (!request) return;


      const updatedRequest = {
        ...request,
        status: newStatus,
        updated_at: new Date().toISOString()
      };


      const result = await window.dataSdk.update(updatedRequest);
      if (result.isOk) {
        showToast(`Request ${newStatus.toLowerCase()}!`);
      } else {
        showToast('Failed to update request.');
      }
    }
